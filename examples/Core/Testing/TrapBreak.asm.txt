; TrapBreak: Test case for trap break.
; The program contains a long-running loop, containing a trap break.
; Sigma16: https://jtod.github.io/home/Sigma16  John O'Donnell, 2023


; sum := 0
; for i := 0 to n
;   sum := sum + 1

; R1 = constant 1
; R2 = n  number of iterations for outer loop
; R3 = sum
; R4 = i
; R5 = temp

      lea    R1,1       ; R1 := 1
      load   R2,n       ; R2 := n
      add    R3,R0,R0   ; sum := 0

      add    R4,R0,R0   ; i := 0
loop
      lea    R5,break   ; R5 := trap code for break
      trap   R5,R0,R0   ; break
      cmp    R4,R2      ; compare i with  n
      jumpge done       ; if i >= n then goto done
      add    R3,R3,R1   ; sum := sum + 1
      add    R4,R4,R1   ; i := i + 1
      jump   loop       ; goto loop
done  store  R3,sum     ; store sum
      trap   R0,R0,R0   ; terminate      

break equ    4          ; trap code for break
n     data   5          ; number of iterations
sum   data   0
